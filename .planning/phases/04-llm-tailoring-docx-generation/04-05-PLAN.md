**async def get_queued_jobs(session) -> list[Job]:**
- Query jobs with status="matched" (ready for tailoring).
- Order by score DESC (highest match first).
- Return list.

**async def get_next_version(session, job_id: int) -> int:**
- Count existing TailoringRecord rows for this job_id.
- Return count + 1.

**async def save_tailoring_record(session, job_id: int, version: int, intensity: str, base_resume_path: str, tailored_resume_path: str | None, cover_letter_path: str | None, result: "TailoringResult", status: str) -> TailoringRecord:**
- Create and persist a TailoringRecord with all fields from the TailoringResult.
- Store validation_warnings as JSON string.
- Store total token counts and estimated cost.
- Compute prompt_hash as SHA256 of the system prompt + resume text + job description.
- Return the saved record.

**async def save_cost_entries(session, record_id: int, llm_calls: list[dict]):**
- For each call in llm_calls, create a CostLedger row.
- Each entry has: tailoring_record_id, call_type, model, token counts, cost, month.

**async def get_tailoring_records_for_job(session, job_id: int) -> list[TailoringRecord]:**
- Query all TailoringRecord rows for a job, ordered by version.

**async def get_latest_tailoring(session, job_id: int) -> TailoringRecord | None:**
- Return the highest-version completed TailoringRecord for a job, or None.

**async def get_monthly_cost_summary(session) -> dict:**
- Query CostLedger for current month.
- Return {"total_cost": float, "call_count": int, "by_type": {"tailor": float, "validate": float, "cover_letter": float}}.

**Artifact path helpers:**

def artifact_dir(job_id: int) -> Path:
    """Return data/resumes/{job_id}/, creating if needed."""
    d = Path(get_settings().data_dir) / "resumes" / str(job_id)
    d.mkdir(parents=True, exist_ok=True)
    return d

def cover_letter_artifact_path(job_id: int, version: int) -> Path:
    return artifact_dir(job_id) / f"cover_letter_v{version}.docx"

**Create `app/tailoring/pipeline.py`:**

`async def run_tailoring(ctx: RunContext, session_factory) -> dict:`

1. Load dependencies:
   - Get base resume path via `get_resume_path()`. If None, log and return early (no resume uploaded).
   - Extract resume sections via `extract_resume_text()`.
   - Get Settings row for tailoring_intensity default.

2. Get queued jobs:
   ```python
   async with session_factory() as session:
       jobs = await get_queued_jobs(session)
   ```
   If no jobs, return early.

3. Create provider and budget guard:
   ```python
   async with session_factory() as session:
       provider = await get_provider(session)
   budget = BudgetGuard()
   ```

4. Process each job:
   ```python
   tailored_count = 0
   failed_count = 0
   budget_halted = False
   
   for job in jobs:
       # Kill-switch check between jobs
       await ctx.killswitch_check()  # or however the existing pattern works
       
       # Budget check
       async with session_factory() as session:
           can_proceed, spent, cap, is_warning = await budget.check_budget(session)
       
       if not can_proceed:
           budget_halted = True
           log.warning("tailoring_budget_halt", spent=spent, cap=cap, job_id=job.id)
           break
       
       if is_warning:
           log.warning("tailoring_budget_warning", spent=spent, cap=cap, threshold="80%")
       
       # Get version number
       async with session_factory() as session:
           version = await get_next_version(session, job.id)
       
       # Compute artifact paths
       resume_path = resume_artifact_path(job.id, version)
       cl_path = cover_letter_artifact_path(job.id, version)
       
       # Run tailoring engine
       try:
           result = await tailor_resume(
               provider=provider,
               resume_sections=resume_sections["sections"],
               job_description=job.description,
               intensity=settings.tailoring_intensity,
           )
       except Exception as e:
           log.exception("tailoring_error", job_id=job.id, error=str(e))
           # Save failed record
           async with session_factory() as session:
               await save_tailoring_record(session, job.id, version, settings.tailoring_intensity,
                   str(base_resume_path), None, None, result=None, status="failed")
               # Update job status
               job_row = await session.get(Job, job.id)
               if job_row:
                   job_row.status = "failed"
               await session.commit()
           failed_count += 1
           continue
       
       if result.success:
           # Write DOCX artifacts
           build_tailored_docx(base_resume_path, result.tailored_sections, resume_path)
           if result.cover_letter_paragraphs:
               build_cover_letter_docx(result.cover_letter_paragraphs, cl_path, base_resume_path)
           
           # Save record and costs
           async with session_factory() as session:
               record = await save_tailoring_record(session, job.id, version,
                   settings.tailoring_intensity, str(base_resume_path),
                   str(resume_path), str(cl_path) if result.cover_letter_paragraphs else None,
                   result=result, status="completed")
               await save_cost_entries(session, record.id, result.llm_calls)
               
               # Debit budget
               total_cost = sum(c.get("cost", 0) for c in result.llm_calls)
               await budget.debit(session, total_cost, record.id, "tailor",
                   "claude-sonnet-4-5", result.total_input_tokens,
                   result.total_output_tokens, result.total_cache_read_tokens,
                   result.total_cache_write_tokens)
               
               # Update job status
               job_row = await session.get(Job, job.id)
               if job_row:
                   job_row.status = "tailored"
               await session.commit()
           
           tailored_count += 1
       else:
           # Validation failed after all retries
           async with session_factory() as session:
               record = await save_tailoring_record(session, job.id, version,
                   settings.tailoring_intensity, str(base_resume_path),
                   None, None, result=result, status="rejected")
               await save_cost_entries(session, record.id, result.llm_calls)
               
               # Still debit budget for failed attempts
               total_cost = sum(c.get("cost", 0) for c in result.llm_calls)
               await budget.debit(session, total_cost, record.id, "tailor",
                   "claude-sonnet-4-5", result.total_input_tokens,
                   result.total_output_tokens, result.total_cache_read_tokens,
                   result.total_cache_write_tokens)
               
               job_row = await session.get(Job, job.id)
               if job_row:
                   job_row.status = "failed"
               await session.commit()
           
           failed_count += 1
   ```

5. Return counts: `{"tailored": tailored_count, "failed": failed_count, "budget_halted": budget_halted}`
