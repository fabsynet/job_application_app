**`app/manual_apply/fetcher.py`**: async httpx-based fetcher with structured error class. Follow research Pattern 8 verbatim. Reuse `detect_source` and `strip_html` from `app.discovery.fetchers`. Key behaviours:

```python
from __future__ import annotations

import re
from dataclasses import dataclass
from urllib.parse import urlparse

import httpx
import structlog

from app.discovery.fetchers import detect_source, strip_html

log = structlog.get_logger(__name__)

USER_AGENT = "Mozilla/5.0 (compatible; JobApplyBot/1.0; +local)"


@dataclass
class ParsedJob:
    title: str
    company: str
    description: str
    description_html: str
    url: str
    source: str  # 'greenhouse' | 'lever' | 'ashby' | 'manual'
    external_id: str


class FetchError(Exception):
    def __init__(self, reason: str, status: int | None = None):
        super().__init__(reason)
        self.reason = reason
        self.status = status


async def fetch_and_parse(url: str) -> ParsedJob:
    """Fetch a job URL and return a ParsedJob. Raises FetchError on any failure.

    Routing:
      1. Call detect_source(url). If known ATS (greenhouse/lever/ashby),
         delegate to the matching Phase 3 normalizer by fetching + parsing
         the ATS-specific payload.
      2. Otherwise, GET the URL, strip HTML, best-effort guess title/company.
    """
    try:
        source_type, slug = detect_source(url)
    except Exception:
        source_type, slug = "manual", ""

    timeout = httpx.Timeout(10.0, connect=5.0)
    headers = {"User-Agent": USER_AGENT, "Accept": "text/html,application/json"}

    async with httpx.AsyncClient(timeout=timeout, follow_redirects=True, headers=headers) as client:
        try:
            resp = await client.get(url)
        except httpx.TimeoutException as exc:
            raise FetchError("timeout") from exc
        except httpx.ConnectError as exc:
            raise FetchError(f"connect_failed: {exc}") from exc
        except httpx.RequestError as exc:
            raise FetchError(f"request_error: {exc}") from exc

    if resp.status_code == 404:
        raise FetchError("not_found", status=404)
    if resp.status_code in (401, 403):
        raise FetchError("auth_wall", status=resp.status_code)
    if resp.status_code >= 400:
        raise FetchError(f"http_{resp.status_code}", status=resp.status_code)

    body = resp.text or ""

    # Known ATS: the detect_source result is authoritative. For v1, DON'T
    # round-trip through the discovery normalizer API (too complex). Instead,
    # parse the same HTML body as generic + stamp source_type.
    if source_type in ("greenhouse", "lever", "ashby"):
        return _best_effort_parse(url, body, source_type)

    # Bot-wall edge cases: empty body or obvious auth page -> raise auth_wall
    if len(body.strip()) < 200:
        raise FetchError("empty_body", status=resp.status_code)

    return _best_effort_parse(url, body, "manual")


def _best_effort_parse(url: str, body: str, source: str) -> ParsedJob:
    """Parse a generic HTML page — <title>, og:site_name, strip_html for description."""
    # title
    m = re.search(r"<title[^>]*>(.*?)</title>", body, re.I | re.S)
    title = (m.group(1).strip() if m else "").split(" at ")[0].strip() or "Unknown Role"

    # company: try og:site_name, fall back to hostname
    m = re.search(
        r'<meta[^>]+property=["\']og:site_name["\'][^>]+content=["\']([^"\']+)',
        body, re.I,
    )
    if m:
        company = m.group(1).strip()
    else:
        company = urlparse(url).hostname or "Unknown"

    # description: strip all HTML to text (reuses Phase 3 helper)
    description = strip_html(body)[:20000]  # cap at 20k chars for budget safety

    # external_id: hash of URL path component
    external_id = urlparse(url).path or url

    return ParsedJob(
        title=title,
        company=company,
        description=description,
        description_html=body,
        url=url,
        source=source,
        external_id=external_id,
    )


__all__ = ["fetch_and_parse", "ParsedJob", "FetchError"]
```

**`app/manual_apply/service.py`**: dedup-respecting job creation.

```python
from __future__ import annotations

from sqlalchemy.ext.asyncio import AsyncSession
from sqlmodel import select

from app.discovery.models import Job
from app.discovery.scoring import job_fingerprint
from app.discovery.service import get_job_by_fingerprint
from app.manual_apply.fetcher import ParsedJob


async def check_duplicate(session: AsyncSession, parsed: ParsedJob) -> Job | None:
    fp = job_fingerprint(parsed.url, parsed.title, parsed.company)
    return await get_job_by_fingerprint(session, fp)


async def create_manual_job(session: AsyncSession, parsed: ParsedJob) -> Job:
    """Create a Job row for a manually-pasted URL.

    - Reuses canonical job_fingerprint for dedup (research pitfall 6).
    - Bypasses keyword-match threshold: status is set directly to 'matched'
      so the next tailoring run picks it up (MANL-04).
    - Caller is responsible for calling check_duplicate() first if it wants
      to return the existing job instead of raising.
    """
    fp = job_fingerprint(parsed.url, parsed.title, parsed.company)
    existing = await get_job_by_fingerprint(session, fp)
    if existing is not None:
        return existing  # idempotent — don't create a duplicate
    job = Job(
        fingerprint=fp,
        external_id=parsed.external_id,
        title=parsed.title,
        company=parsed.company,
        description=parsed.description,
        description_html=parsed.description_html,
        url=parsed.url,
        source=parsed.source,
        score=100,  # manual opt-in, forced match
        matched_keywords="manual_paste",
        status="matched",  # bypasses threshold, enters tailoring pipeline
    )
    session.add(job)
    await session.commit()
    await session.refresh(job)
    return job
```

**Router** `app/web/routers/manual_apply.py` with four endpoints. Follow the `app/web/routers/jobs.py` shape.

| Method | Path | Behavior |
|---|---|---|
| GET  | `/manual-apply`          | Renders `manual_apply/index.html.j2` — paste input |
| POST | `/manual-apply/preview`  | Form `url` → calls `fetch_and_parse`. On success → `_preview.html.j2` fragment. On `FetchError` → `_fallback.html.j2` with the error reason prefilled |
| POST | `/manual-apply/confirm`  | Form `title/company/description/source/url` → `check_duplicate` → if exists, return a fragment with "Already exists as job #N" + link to `/jobs/{id}`; else build a `ParsedJob` + `create_manual_job`, return a success fragment with link to `/review/{new_id}` or `/jobs/{new_id}` |
| POST | `/manual-apply/fallback` | Direct-entry path from `_fallback.html.j2` — same body as /confirm but no URL-fetch step |

Both `confirm` and `fallback` endpoints must call `create_manual_job` which hits the dedup guard.
