```python
"""Applied-jobs service â€” counts, list, detail, download helpers.

All downstream of 05-01's state machine constants and 05-04's Submission
rows. No writes â€” this module is read-only for the dashboard.
"""
from __future__ import annotations

from dataclasses import dataclass
from datetime import datetime, timedelta
from pathlib import Path
from typing import Any

from sqlalchemy import func
from sqlalchemy.ext.asyncio import AsyncSession
from sqlmodel import select

from app.discovery.models import Job
from app.submission.models import Submission
from app.tailoring.models import TailoringRecord

APPLIED_SORT_COLUMNS: dict[str, Any] = {
    "company": Job.company,
    "title": Job.title,
    "score": Job.score,
    "submitted_at": Submission.sent_at,
    "status": Job.status,
    "source": Job.source,
}


@dataclass
class StateCounts:
    submitted: int = 0
    approved: int = 0
    failed: int = 0
    skipped: int = 0
    needs_info: int = 0
    tailored: int = 0


async def state_counts_for_window(
    session: AsyncSession,
    *,
    since: datetime,
) -> StateCounts:
    """Counts per Job.status for jobs whose first_seen_at >= since.

    'Today' = since=midnight-local; 'Last 7 days' = since=now-7d.
    """
    rows = (
        await session.execute(
            select(Job.status, func.count(Job.id))
            .where(Job.first_seen_at >= since)
            .group_by(Job.status)
        )
    ).all()
    counts = StateCounts()
    for status, count in rows:
        if hasattr(counts, status):
            setattr(counts, status, count)
    return counts


async def list_applied_jobs(
    session: AsyncSession,
    *,
    sort_by: str = "submitted_at",
    sort_dir: str = "desc",
    status_filter: list[str] | None = None,
    source_filter: list[str] | None = None,
    limit: int = 50,
    offset: int = 0,
) -> tuple[list[tuple[Job, Submission | None, TailoringRecord | None]], int]:
    """Returns (rows, total).

    Includes BOTH `submitted` jobs (which have Submission rows) and
    `approved` jobs (which do not yet â€” manual-completion path per SC-5).
    """
    # Default filter: any state that represents "the user acted on this job"
    default_states = ["submitted", "approved", "failed", "skipped", "needs_info"]
    filter_states = status_filter or default_states

    # LEFT OUTER JOIN to Submission so approved-but-unsent jobs still appear.
    # Latest submission per job via a subquery.
    latest_sub_subq = (
        select(Submission.job_id, func.max(Submission.id).label("max_id"))
        .group_by(Submission.job_id)
        .subquery()
    )
    q = (
        select(Job, Submission, TailoringRecord)
        .join(latest_sub_subq, latest_sub_subq.c.job_id == Job.id, isouter=True)
        .join(Submission, Submission.id == latest_sub_subq.c.max_id, isouter=True)
        .join(TailoringRecord, TailoringRecord.job_id == Job.id, isouter=True)
        .where(Job.status.in_(filter_states))
    )
    if source_filter:
        q = q.where(Job.source.in_(source_filter))

    col = APPLIED_SORT_COLUMNS.get(sort_by, Submission.sent_at)
    q = q.order_by(col.desc() if sort_dir == "desc" else col.asc())
    q = q.limit(limit).offset(offset)

    rows = (await session.execute(q)).all()
    # total count (separate query is fine at v1 scale)
    total = (
        await session.execute(
            select(func.count(Job.id)).where(Job.status.in_(filter_states))
        )
    ).scalar_one()
    return [(r[0], r[1], r[2]) for r in rows], total


async def get_applied_detail(session: AsyncSession, job_id: int) -> dict:
    """Detail view â€” full JD, latest tailoring artifact paths, submission info."""
    job = await session.get(Job, job_id)
    if job is None:
        return {}
    # Latest record (reuse existing helper)
    from app.tailoring.service import get_latest_tailoring
    record = await get_latest_tailoring(session, job_id)
    # Latest submission (may be None for approved-but-unsent)
    latest = (
        await session.execute(
            select(Submission).where(Submission.job_id == job_id)
            .order_by(Submission.id.desc()).limit(1)
        )
    ).scalar_one_or_none()

    cover_letter_text = ""
    if record is not None and record.cover_letter_path:
        from app.submission.builder import extract_cover_letter_plaintext
        try:
            cover_letter_text = extract_cover_letter_plaintext(record.cover_letter_path)
        except Exception:
            cover_letter_text = "(unable to read cover letter file)"

    # DOCX preview HTML via existing Phase 4 helper
    tailored_preview_html = ""
    if record is not None and record.tailored_resume_path:
        from app.tailoring.preview import docx_to_html
        try:
            tailored_preview_html = docx_to_html(Path(record.tailored_resume_path))
        except Exception:
            tailored_preview_html = "<p>(preview unavailable)</p>"

    return {
        "job": job,
        "record": record,
        "submission": latest,
        "cover_letter_text": cover_letter_text,
        "tailored_preview_html": tailored_preview_html,
    }


def applied_artifact_paths(record: TailoringRecord | None) -> dict[str, Path | None]:
    """Return {'resume': Path|None, 'cover_letter': Path|None}."""
    if record is None:
        return {"resume": None, "cover_letter": None}
    return {
        "resume": Path(record.tailored_resume_path) if record.tailored_resume_path else None,
        "cover_letter": Path(record.cover_letter_path) if record.cover_letter_path else None,
    }


__all__ = [
    "APPLIED_SORT_COLUMNS",
    "StateCounts",
    "state_counts_for_window",
    "list_applied_jobs",
    "get_applied_detail",
    "applied_artifact_paths",
]
```
