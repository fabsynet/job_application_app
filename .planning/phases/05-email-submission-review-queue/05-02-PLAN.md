```python
"""Phase 5 email builder — pure helpers, no I/O except DOCX read."""
from __future__ import annotations

import re
from email.message import EmailMessage
from pathlib import Path

# Non-noreply email regex (order matters: filters first)
_EMAIL_RE = re.compile(r"[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}")
_NOREPLY_PREFIXES = ("noreply", "no-reply", "donotreply", "do-not-reply", "notifications")

_DOCX_MAINTYPE = "application"
_DOCX_SUBTYPE = "vnd.openxmlformats-officedocument.wordprocessingml.document"


def _slug_ascii(value: str) -> str:
    """Strict ASCII slug for filename components: `[A-Za-z0-9_]` only."""
    cleaned = (value or "").strip().replace(" ", "_")
    slug = re.sub(r"[^A-Za-z0-9_]", "", cleaned)
    return slug or "Unknown"


def build_subject(*, role: str, company: str) -> str:
    """Locked format: `Application for {role} at {company}`."""
    role = (role or "Application").strip() or "Application"
    company = (company or "Unknown").strip() or "Unknown"
    return f"Application for {role} at {company}"


def build_attachment_filename(*, full_name: str, company: str) -> str:
    """Locked format: `{FullName}_{Company}_Resume.docx`, strict ASCII."""
    return f"{_slug_ascii(full_name)}_{_slug_ascii(company)}_Resume.docx"


def extract_cover_letter_plaintext(cover_letter_path: Path | str) -> str:
    """Read the cover letter DOCX and return paragraph text joined by blank lines.

    Phase 4 only persists the cover letter as a DOCX at
    `TailoringRecord.cover_letter_path`. There is no plaintext field.
    Re-extract via python-docx and join with `\\n\\n` — single `\\n` collapses
    paragraph breaks (research pitfall 10).
    """
    from docx import Document
    doc = Document(str(cover_letter_path))
    paragraphs = [p.text for p in doc.paragraphs if p.text and p.text.strip()]
    return "\n\n".join(paragraphs)


def resolve_recipient_email(job_description: str) -> str | None:
    """Regex the first non-noreply email out of the job description.

    Returns None if no suitable address is found. Callers should flip
    `Job.status = 'needs_info'` on None (research open question #1).
    """
    if not job_description:
        return None
    for match in _EMAIL_RE.finditer(job_description):
        addr = match.group(0)
        local = addr.split("@", 1)[0].lower()
        if any(local.startswith(p) for p in _NOREPLY_PREFIXES):
            continue
        return addr
    return None


def build_email_message(
    *,
    from_addr: str,
    to_addr: str,
    subject: str,
    body_text: str,
    attachment_path: Path | str,
    attachment_filename: str,
) -> EmailMessage:
    """Construct an EmailMessage with a single DOCX attachment.

    Uses the modern stdlib API (`EmailMessage.add_attachment`) — handles
    Base64 + RFC 2231 filename encoding automatically. Do NOT use
    MIMEMultipart/MIMEBase (legacy, header-encoding bugs).
    """
    msg = EmailMessage()
    msg["From"] = from_addr
    msg["To"] = to_addr
    msg["Subject"] = subject
    msg.set_content(body_text)  # plain text, preserves newlines
    data = Path(attachment_path).read_bytes()
    msg.add_attachment(
        data,
        maintype=_DOCX_MAINTYPE,
        subtype=_DOCX_SUBTYPE,
        filename=attachment_filename,
    )
    return msg


__all__ = [
    "build_subject",
    "build_attachment_filename",
    "extract_cover_letter_plaintext",
    "resolve_recipient_email",
    "build_email_message",
]
```

```python
"""Phase 5 SMTP credential loader — reads Secret rows, decrypts via Fernet."""
from __future__ import annotations

from dataclasses import dataclass
from sqlalchemy.ext.asyncio import AsyncSession
from sqlmodel import select

from app.config import get_settings as get_app_settings
from app.db.models import Secret
from app.security.fernet import FernetVault


class SmtpCredsMissing(Exception):
    """Raised when a required smtp_* Secret row is absent."""
    def __init__(self, name: str):
        super().__init__(f"SMTP credential missing: {name}")
        self.name = name


@dataclass(frozen=True)
class SmtpCreds:
    host: str
    port: int
    username: str
    password: str


_REQUIRED = ("smtp_host", "smtp_port", "smtp_user", "smtp_password")


async def load_smtp_creds(session: AsyncSession) -> SmtpCreds:
    vault = FernetVault.from_env(get_app_settings().fernet_key)
    raw: dict[str, str] = {}
    for name in _REQUIRED:
        row = (
            await session.execute(select(Secret).where(Secret.name == name))
        ).scalar_one_or_none()
        if row is None:
            raise SmtpCredsMissing(name)
        raw[name] = vault.decrypt(row.ciphertext)
    # Research pitfall 7: smtp_port is written as a string, must int() before use.
    return SmtpCreds(
        host=raw["smtp_host"],
        port=int(raw["smtp_port"]),
        username=raw["smtp_user"],
        password=raw["smtp_password"],
    )


__all__ = ["SmtpCreds", "SmtpCredsMissing", "load_smtp_creds"]
```

```python
"""Phase 5 SMTP sender — async aiosmtplib wrapper with structured failure."""
from __future__ import annotations

from dataclasses import dataclass
from email.message import EmailMessage

import aiosmtplib
import structlog

log = structlog.get_logger(__name__)


class SubmissionSendError(Exception):
    """Wraps any aiosmtplib failure with a stable error_class name for signatures."""
    def __init__(self, *, error_class: str, message: str, cause: Exception | None = None):
        super().__init__(message)
        self.error_class = error_class
        self.message = message
        self.cause = cause


@dataclass
class SmtpConfig:
    host: str
    port: int
    username: str
    password: str
    timeout: float = 30.0


async def send_via_smtp(msg: EmailMessage, cfg: SmtpConfig) -> None:
    """Send an EmailMessage via aiosmtplib.send.

    STARTTLS on 587 (standard submission port); implicit TLS on 465.
    Any port outside these two uses neither (plain — dev/test only).

    Raises SubmissionSendError with a stable error_class the failure-suppression
    table can hash on.
    """
    try:
        await aiosmtplib.send(
            msg,
            hostname=cfg.host,
            port=cfg.port,
            username=cfg.username,
            password=cfg.password,
            start_tls=(cfg.port == 587),
            use_tls=(cfg.port == 465),
            timeout=cfg.timeout,
        )
    except aiosmtplib.SMTPAuthenticationError as exc:
        raise SubmissionSendError(
            error_class="SMTPAuthenticationError",
            message=str(exc),
            cause=exc,
        ) from exc
    except aiosmtplib.SMTPRecipientsRefused as exc:
        raise SubmissionSendError(
            error_class="SMTPRecipientsRefused",
            message=str(exc),
            cause=exc,
        ) from exc
    except aiosmtplib.SMTPServerDisconnected as exc:
        raise SubmissionSendError(
            error_class="SMTPServerDisconnected",
            message=str(exc),
            cause=exc,
        ) from exc
    except aiosmtplib.SMTPTimeoutError as exc:
        raise SubmissionSendError(
            error_class="SMTPTimeoutError",
            message=str(exc),
            cause=exc,
        ) from exc
    except aiosmtplib.SMTPException as exc:
        raise SubmissionSendError(
            error_class=type(exc).__name__,
            message=str(exc),
            cause=exc,
        ) from exc


__all__ = ["SmtpConfig", "SubmissionSendError", "send_via_smtp"]
```

```python
from app.submission.sender import send_via_smtp, SmtpConfig, SubmissionSendError


class _FakeSend:
    def __init__(self, exc: Exception | None = None):
        self.exc = exc
        self.calls: list[dict] = []

    async def __call__(self, msg, **kwargs):
        self.calls.append({"msg": msg, **kwargs})
        if self.exc is not None:
            raise self.exc


@pytest.mark.asyncio
async def test_send_uses_starttls_on_587(monkeypatch):
    fake = _FakeSend()
    monkeypatch.setattr(aiosmtplib, "send", fake)
    cfg = SmtpConfig(host="smtp.example.com", port=587, username="u", password="p")
    await send_via_smtp(EmailMessage(), cfg)
    assert fake.calls[0]["start_tls"] is True
    assert fake.calls[0]["use_tls"] is False
    assert fake.calls[0]["port"] == 587
```
